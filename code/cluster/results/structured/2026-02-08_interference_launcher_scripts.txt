#!/usr/bin/env bash
set -euo pipefail

REPO=/home/ubuntu/ai-performance-engineering/code/cluster
cd "$REPO"

RUN_PREFIX=2026-02-08_smoke_high_impact_local_retry
LOOP_LOG=results/raw/${RUN_PREFIX}_loop.log
STATUS_FILE=results/raw/${RUN_PREFIX}_loop.status
MAX_ATTEMPTS=12

mkdir -p results/raw results/structured docs/figures
: > "$LOOP_LOG"
rm -f "$STATUS_FILE"

log() {
  echo "[$(date -u +%FT%TZ)] $*" | tee -a "$LOOP_LOG"
}

is_idle() {
  local suites locks gpu_count vllm_count
  suites="$(pgrep -af '^bash scripts/run_cluster_eval_suite.sh' || true)"
  locks="$(pgrep -af '^bash /home/ubuntu/ai-performance-engineering/code/cluster/scripts/run_with_gpu_clocks.sh' || true)"
  vllm_count="$(ps -eo pid,args | awk '$2 ~ /python/ && $0 ~ /\/usr\/local\/bin\/vllm serve/ {c++} END{print c+0}')"
  gpu_count="$(nvidia-smi --query-compute-apps=pid --format=csv,noheader 2>/dev/null | sed '/^$/d' | wc -l | tr -d ' ')"

  if [[ -z "$suites" && -z "$locks" && "$vllm_count" == "0" && "$gpu_count" == "0" ]]; then
    return 0
  fi

  log "BUSY suites=$( [[ -n "$suites" ]] && echo 1 || echo 0 ) locks=$( [[ -n "$locks" ]] && echo 1 || echo 0 ) vllm=${vllm_count} gpu_procs=${gpu_count}"
  if [[ -n "$suites" ]]; then
    while IFS= read -r line; do
      [[ -z "$line" ]] || log "  suite: $line"
    done <<< "$suites"
  fi
  if [[ -n "$locks" ]]; then
    local i=0
    while IFS= read -r line; do
      [[ -z "$line" ]] && continue
      log "  lock: $line"
      i=$((i+1))
      [[ "$i" -ge 4 ]] && break
    done <<< "$locks"
  fi
  return 1
}

attempt=1
while (( attempt <= MAX_ATTEMPTS )); do
  log "ATTEMPT ${attempt}/${MAX_ATTEMPTS}"

  until is_idle; do
    sleep 20
  done
  log "IDLE window confirmed"

  run_id="${RUN_PREFIX}${attempt}"
  port=$((18088 + attempt))
  driver_log="results/raw/${run_id}_driver.log"

  log "RUN start run_id=${run_id} port=${port}"
  set +e
  timeout 90m bash scripts/run_cluster_eval_suite.sh \
    --run-id "$run_id" \
    --hosts localhost \
    --labels node1 \
    --skip-bootstrap-nodes \
    --disable-fp4 \
    --health-suite off \
    --model Qwen/Qwen2.5-0.5B-Instruct \
    --tp 1 \
    --isl 128 \
    --osl 128 \
    --concurrency-range "1 2" \
    --port "$port" \
    --fio-runtime 10 \
    --enable-mamf --mamf-mode quick --mamf-concurrent \
    --enable-allreduce-stability --allreduce-payload-gib 0.5 --allreduce-iters 40 --allreduce-warmup 5 \
    --enable-allreduce-latency-comp --allreduce-latency-payload-gib 1.0 --allreduce-latency-chunks 128 --allreduce-latency-iters 3 --allreduce-latency-warmup 1 \
    --enable-allgather-control-plane --allgather-control-iters 500 --allgather-control-warmup 50 \
    --enable-nccl-algo-comparison --nccl-algos Ring,Tree,NVLS,auto \
    > "$driver_log" 2>&1
  rc=$?
  set -e

  log "RUN end run_id=${run_id} rc=${rc}"

  if [[ "$rc" -eq 0 ]]; then
    echo "success ${run_id}" > "$STATUS_FILE"
    log "SUCCESS run_id=${run_id}"
    exit 0
  fi

  attempt=$((attempt + 1))
  sleep 10
done

echo "failed attempts=${MAX_ATTEMPTS}" > "$STATUS_FILE"
log "FAILED all attempts exhausted"
exit 1

run_step() {
  local name="$1"
  shift
  local start end rc log_path
  log_path="${SUITE_LOG_DIR}/${name}.log"
  start="$(date -Iseconds)"
  echo "==> [${start}] START ${name}"
  echo "    cmd: $*"
  echo "    log: ${log_path}"
  set +e
  "$@" >"$log_path" 2>&1
  rc=$?
  set -e
  end="$(date -Iseconds)"
  echo "<== [${end}] END ${name} rc=${rc}"
  if [[ "$rc" -ne 0 ]]; then
    fail=1
    echo "---- ${name} (last 80 lines) ----" >&2
    tail -n 80 "$log_path" >&2 || true

  return 0
